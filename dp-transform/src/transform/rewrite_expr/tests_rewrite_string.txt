$ desugars fstring

f"x={x}"
=
getattr("", "join")(("x=", __dp__.builtins.format(x)))

$ desugars tstring

t"x={x}"
=
__dp__.template(*("x=", __dp__.interpolation(x, "x", None, "")))

$ preserves conversions and format specifiers

f"{value!r}:{other:0.2f}"
=
getattr("", "join")((__dp__.builtins.format(__dp__.builtins.repr(value)), ":", __dp__.builtins.format(other, "0.2f")))

$ fstring uses builtins format when shadowed

format = 1
value = "x"
result = f"{value}"
=
format = 1
value = "x"
result = __dp__.builtins.format(value)

$ fstring debug uses repr and source text

f"{x=}"
=
getattr("", "join")(("x=", __dp__.builtins.format(__dp__.builtins.repr(x))))
